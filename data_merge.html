DM_cg02813121<-subset(DM_S100A8,cgnumber=="cg02813121")[,-1]
GE_DM<-merge(DM_cg02813121,GE_S100A8,by="sampleID",all=T)
GE_DM$condition<-ifelse(grepl("A",GE_DM$sampleID),"H","UC")
